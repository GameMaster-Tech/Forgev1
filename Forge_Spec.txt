### 2.3 Feature 3: Operational Invariant Assertions (The Workspace Unit Tester)

If an entry fails verification, the workspace blocks the execution, surfaces the breakdown node on the timeline, and provides structural correction paths.

### 2.4 Feature 4: Tempo Engine (Native Calendar Auto-Sorting)

* **The Execution:** Once a simulated sandbox branch satisfies all operational invariant conditions and is accepted by the user, the Tempo Engine loops through the delta graph, programmatically executing micro-adjustments across your native timeline, re-allocating meeting slots, and compressing dead administrative gaps.

export class ForgeSyncCompiler {
  private activeGraph: Map<NodeId, ForgeGraphNode>;
  private invariants: WorkspaceInvariant[];

  constructor(initialGraph: Map<NodeId, ForgeGraphNode>, invariants: WorkspaceInvariant[]) {
    this.activeGraph = initialGraph;
    this.invariants = invariants;
  }

  public generateImpactReport(targetNodeId: NodeId, proposedDelta: Record<string, any>): VisualDeltaMap {
    const deltaMap: VisualDeltaMap = {
      scenarioPrompt: `Simulation for modification on Node: ${targetNodeId}`,
      isViable: true,
      globalRiskScore: 0,
      mutations: [],
      assertionFailures: []
    };

    const sandboxGraph = new Map<NodeId, ForgeGraphNode>();
    this.activeGraph.forEach((val, key) => {
      sandboxGraph.set(key, {
        ...val,
        payload: { ...val.payload, metadata: { ...val.payload.metadata } },
        upstreamDependencies: [...val.upstreamDependencies],
        downstreamDependencies: [...val.downstreamDependencies]
      });
    });

    const targetNode = sandboxGraph.get(targetNodeId);
    if (targetNode) {
      const prevMeta = { ...targetNode.payload.metadata };
      targetNode.payload.metadata = { ...targetNode.payload.metadata, ...proposedDelta };
      
      Object.keys(proposedDelta).forEach(key => {
        deltaMap.mutations.push({
          nodeId: targetNodeId,
          targetField: `metadata.${key}`,
          previousValue: prevMeta[key],
          proposedValue: proposedDelta[key],
          deltaMagnitude: "Direct input manipulation"
        });
      });
    }

    const queue: NodeId[] = targetNode ? [...targetNode.downstreamDependencies] : [];
    const visited = new Set<NodeId>();

    while (queue.length > 0) {
      const currentId = queue.shift()!;
      if (visited.has(currentId)) continue;
      visited.add(currentId);

      const node = sandboxGraph.get(currentId);
      if (!node) continue;

      if (node.category === ForgeNodeCategory.CALENDAR_EVENT || node.category === ForgeNodeCategory.TASK) {
        const originalStart = node.payload.metadata.startDate ? new Date(node.payload.metadata.startDate.getTime()) : null;
        if (originalStart && proposedDelta.daysShift) {
          const newStart = new Date(originalStart.getTime() + proposedDelta.daysShift * 24 * 60 * 60 * 1000);
          node.payload.metadata.startDate = newStart;
          
          deltaMap.mutations.push({
            nodeId: currentId,
            targetField: 'metadata.startDate',
            previousValue: originalStart,
            proposedValue: newStart,
            deltaMagnitude: `${proposedDelta.daysShift > 0 ? '+' : ''}${proposedDelta.daysShift} days shift`
          });
        }
      }

      node.downstreamDependencies.forEach(id => queue.push(id));
    }

    for (const invariant of this.invariants) {
      const result = invariant.evaluator(sandboxGraph);
      if (!result.passed) {
        deltaMap.assertionFailures.push({
          invariantId: invariant.id,
          description: invariant.description,
          suggestedFix: result.errorDetail || "Constraint limit breached."
        });
      }
    }

    deltaMap.isViable = deltaMap.assertionFailures.length === 0;
    deltaMap.globalRiskScore = deltaMap.isViable ? Math.min(deltaMap.mutations.length * 5, 100) : 100;
    return deltaMap;
  }

  public async evaluateSemanticReactivity(
    newNodeContent: string, 
    embeddingApiProxy: (text: string) => Promise<Float32Array>,
    llmValidationProxy: (proseA: string, proseB: string) => Promise<{ conflict: boolean; reason?: string }>
  ): Promise<Array<{ NodeId; reason: string targetNodeId: }>> {
    const conflicts: Array<{ targetNodeId: NodeId; reason: string }> = [];
    const incomingEmbedding = await embeddingApiProxy(newNodeContent);

    for (const [id, node] of this.activeGraph.entries()) {
      if (node.category === ForgeNodeCategory.PROSE && node.semanticEmbedding) {
        let dotProduct = 0;
        for (let i = 0; i < incomingEmbedding.length; i++) {
          dotProduct += incomingEmbedding[i] * node.semanticEmbedding[i];
        }
        
        if (dotProduct > 0.75) {
          const evaluation = await llmValidationProxy(newNodeContent, node.payload.content);
          if (evaluation.conflict) {
            conflicts.push({
              targetNodeId: id,
              reason: evaluation.reason || "Semantic inconsistency verified by LLM analysis layer."
            });
          }
        }
      }
    }
    return conflicts;
  }
}

export class TempoEngine {
  public executeCalendarSorting(
    graph: Map<NodeId, ForgeGraphNode>, 
    approvedDelta: VisualDeltaMap
  ): Map<NodeId, ForgeGraphNode> {
    const sortedGraph = new Map<NodeId, ForgeGraphNode>(graph);

    approvedDelta.mutations.forEach(mutation => {
      const node = sortedGraph.get(mutation.nodeId);
      if (node) {
        if (mutation.targetField === 'metadata.startDate') {
          node.payload.metadata.startDate = new Date(mutation.proposedValue);
        } else if (mutation.targetField === 'metadata.endDate') {
          node.payload.metadata.endDate = new Date(mutation.proposedValue);
        }
        node.version += 1;
        node.status = 'STABLE';
      }
    });

    const events = Array.from(sortedGraph.values())
      .filter(n => n.category === ForgeNodeCategory.CALENDAR_EVENT && n.payload.metadata.startDate)
      .sort((a, b) => a.payload.metadata.startDate!.getTime() - b.payload.metadata.startDate!.getTime());

    for (let i = 0; i < events.length - 1; i++) {
      const currentEvent = events[i];
      const nextEvent = events[i + 1];
      
      const currentEnd = currentEvent.payload.metadata.endDate || 
        new Date(currentEvent.payload.metadata.startDate!.getTime() + (currentEvent.payload.metadata.durationHours || 1) * 60 * 60 * 1000);

      if (currentEnd.getTime() > nextEvent.payload.metadata.startDate!.getTime()) {
        const overlapDuration = currentEnd.getTime() - nextEvent.payload.metadata.startDate!.getTime();
        nextEvent.payload.metadata.startDate = new Date(nextEvent.payload.metadata.startDate!.getTime() + overlapDuration);
        if (nextEvent.payload.metadata.endDate) {
          nextEvent.payload.metadata.endDate = new Date(nextEvent.payload.metadata.endDate.getTime() + overlapDuration);
        }
        nextEvent.status = 'STABLE';
        nextEvent.version += 1;
      }
    }

    return sortedGraph;
  }
}
